class Solution { public int[] solution(int numer1, int denom1, int numer2, int denom2) { int bunza = 0,bunmo =0, extra = 0; bunza = numer1*denom2 + numer2* denom1; bunmo = denom1*denom2; if(bunza == bunmo){ return new int []{1,1}; }else if(bunza%bunmo ==0){ return new int [] {bunza/bunmo , 1}; } else if(bunza>bunmo){ bunza = bunza%bunmo; extra = (bunza/bunmo)*bunmo; } for(int i = 1 ; i <=bunza ; i++){ while(bunza%i ==0 &&bunmo%i ==0){ bunza = bunza/i; bunmo = bunmo/i; } } return new int []{bunza+extra , bunmo}; } }
